```python
# Zadatak 47
# Саставити програм који за унети природни број
# исписује број чије су цифре у обрнутом редоследу унетог броја.

# Учитавамо број од корисника
# input(): функција која прихвата унос са тастатуре као текст
# int(): функција која конвертује текст у цео број
# n: променљива у којој чувамо унети број
n = int(input("n = "))

# Иницијализујемо променљиву која ће чувати обрнути број
# обрнути: почињемо од 0 и градимо обрнути број цифру по цифру
obrnuti = 0

# Петља која се извршава док год постоје цифре у броју n
# n > 0: услов који проверава да ли је број већи од нуле (тј. да ли још има цифара за обраду)
while n > 0:
    # Додајемо последњу цифру броја n у обрнути број
    # n % 10: операција модуло 10 враћа последњу цифру броја n
    # obrnuti * 10: померамо тренутне цифре у обрнутом броју за једно место улево
    # obrnuti * 10 + n % 10: додајемо последњу цифру броја n на крај обрнутог броја
    obrnuti = obrnuti * 10 + n % 10

    # Уклањамо последњу цифру из оригиналног броја
    # n // 10: целобројно дељење са 10 одбацује последњу цифру броја
    n = n // 10

# Исписујемо резултат
# print(): функција која исписује текст и/или вредности на екран
# Текст "Обрнути =" праћен вредношћу променљиве obrnuti
print("Обрнути =", obrnuti)

```
