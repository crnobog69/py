```python
# Zadatak 38-v
# Саставити програм који употребом WHILE петље исписује:
# в) све бројеве прве десетице у обрнутом редоследу

# Иницијализација бројача 'i' на вредност 10.
# Ово је почетна вредност од које крећемо одбројавање.
i = 10

# WHILE петља која се извршава све док је вредност 'i' већа од 0.
# Услов за петљу: i > 0 - петља ће се извршавати док је вредност променљиве 'i' већа од нуле.
# Ова петља ће се извршити тачно 10 пута (за вредности i од 10 до 1).
while i > 0:
    # Исписивање тренутне вредности променљиве 'i'.
    # print() - уграђена функција која приказује задате вредности на екрану.
    # "i =" - текстуални део који ће бити приказан пре броја.
    # i - вредност променљиве 'i' која ће бити приказана након текста.
    print("i =", i)

    # Смањивање вредности променљиве 'i' за 1.
    # Овај корак је важан јер осигурава да ће петља у неком тренутку завршити
    # (када 'i' достигне вредност 0, услов i > 0 више неће бити испуњен).
    i = i - 1

# Крај програма. Када вредност 'i' постане 0, услов петље више није испуњен
# и програм завршава са извршавањем.
# Резултат: исписани су сви бројеви прве десетице (1-10) у обрнутом редоследу (10-1).

```
