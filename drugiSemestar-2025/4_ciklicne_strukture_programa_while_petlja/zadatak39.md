```python
# Zadatak 39
# Саставити програм за израчунавање суме бројева од 1 до n који су дељиви са 6.
# Број n се уноси са тастатуре

# Учитавање броја n са тастатуре
# Функција input() прихвата унос од корисника као стринг
# Функција int() конвертује стринг у цео број
# Променљива n чува горњу границу опсега бројева
n = int(input("n = "))

# Иницијализација бројача и променљиве за суму
# i је бројач који ће ићи од 1 до n
i = 1
# s је променљива у којој ћемо чувати суму бројева дељивих са 6
s = 0

# Почетак while петље која ће се извршавати док је i мање или једнако n
# while петља је циклична структура која понавља блок кода док је услов испуњен
while i <= n:
    # Провера да ли је тренутни број i дељив са 6
    # Оператор % враћа остатак при дељењу
    # Ако је остатак 0, број је дељив са 6
    if i % 6 == 0:
        # Ако је број дељив са 6, додајемо га на суму
        # НАПОМЕНА: Овде постоји грешка у коду - додаје се 1 уместо вредности i
        s = s + 1
    # Увећавање бројача за 1 после сваке итерације
    i = i + 1

# Испис резултата
# f-string омогућава убацивање променљивих у стринг помоћу {}
# Приказује се коначна сума бројева дељивих са 6 у опсегу од 1 до n
print(f"Сума = {s}")

```
