```python
# Zadatak 40
# Саставити програм којим се:
# а) исписује n елемената Фибоначијевог низа;
# Фибоначијев низ је: f1=1, f2=1, fi=fi-1+fi-2,    i=3, 4, 5, ...

# Учитавамо број елемената Фибоначијевог низа које желимо да испишемо
n = int(input("n = "))

# Иницијализација променљивих:
i = 3  # Бројач - почињемо од трећег елемента (пошто су прва два 1)
prethodni_broj = 1  # f1 = 1 - први број у низу
trenutni_broj = 1  # f2 = 1 - други број у низу

# Исписујемо прва два елемента Фибоначијевог низа
print("ФИБОНАЧИЈЕВИ БРОЈЕВИ: ")
print("f1 = 1 \nf2 = 1")

# Петља за израчунавање и исписивање преосталих елемената
while i <= n:
    # Израчунавамо следећи број у низу као збир претходна два
    sledeci_broj = prethodni_broj + trenutni_broj

    # Померамо вредности за следећу итерацију
    prethodni_broj = trenutni_broj  # Стари тренутни постаје претходни
    trenutni_broj = sledeci_broj  # Нови израчунати број постаје тренутни

    # Исписујемо израчунати број
    print(f"f{i} = {sledeci_broj}")

    # Повећавамо бројач
    i = i + 1

```
