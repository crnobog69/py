```python
# Zadatak 41
# Саставити програм који ће учитавати један за другим низ бројева.
# Крај уноса означен је нулом.
# Наћи и исписати аритметичку средину учитаних бројева,
# узимајући у обзир само оне бројеве који су већи или једнаки од 2 и мањи или једнаки од 6.
# Подразумева се да се уноси бар један број различит од нуле

# Иницијализација променљивих:
# i - бројач који прати колико бројева задовољава услов (2 <= број <= 6)
# s - сума бројева који задовољавају услов (2 <= број <= 6)
i = 0
s = 0

# Исписивање упутства кориснику за унос података
print("Унесите низ реалних бројева или 0 за крај")

# Бесконачна петља која се извршава док корисник не унесе 0
while True:
    # Учитавање реалног броја од корисника
    # float() - функција која конвертује унети текст у реалан број
    # input() - функција која чита унос корисника са тастатуре
    x = float(input())

    # Провера да ли је унесена нула (услов за крај уноса)
    if x == 0:
        # Рачунање аритметичке средине (сума подељена бројем елемената)
        sredina = s / i
        # Исписивање резултата
        # round(sredina, 3) - заокруживање на 3 децимале
        print("Аритметичка средина =", round(sredina, 3))
        # Прекид извршавања петље након исписа резултата
        break

    # Провера да ли број задовољава услов (2 <= број <= 6)
    if x >= 2 and x <= 6:
        # Додавање броја у суму ако задовољава услов
        s = s + x
        # Повећање бројача за 1
        i = i + 1

```
