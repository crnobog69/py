# Zadatak 38-b
# Саставити програм који употребом WHILE петље исписује:
# б) само парне бројеве прве десетице

# Дефинисање почетне вредности бројача
# Почињемо од 2 јер је то први паран број у првој десетици (1-10)
i = 2

# Конструкција WHILE петље која се извршава све док је вредност i мања или једнака 10
# Параметри:
#   - i: бројач који се инкрементира
#   - i <= 10: услов који проверава да ли је бројач мањи или једнак 10
#   - Петља се извршава док је услов тачан (True)
while i <= 10:
    # Исписивање тренутне вредности бројача
    # Параметри функције print():
    #   - "i =": текст који се приказује пре вредности
    #   - i: вредност бројача која се исписује
    # Повратна вредност: Нема (None), функција print() само исписује на стандардни излаз
    print("i =", i)

    # Повећавање вредности бројача за 2
    # Ово обезбеђује да i увек има парну вредност (2, 4, 6, 8, 10)
    # i + 2: израз који додаје 2 на тренутну вредност i
    # i = израз: оператор доделе који ажурира вредност i
    i = i + 2

# Крај програма
# Када i постане 12, услов i <= 10 постаје нетачан и петља се завршава
# Излазни резултат програма: исписани су сви парни бројеви од 1 до 10 (2, 4, 6, 8, 10)
