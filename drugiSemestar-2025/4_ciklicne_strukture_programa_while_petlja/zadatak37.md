```python
# Zadatak 37
# Саставити програм који ће пет пута исписати реченицу 'Поздрав свима!'
# употребом WHILE петље.

# Иницијализација бројача који ће контролисати број понављања петље.
# Почињемо од 1 јер ћемо бројати од 1 до 5.
i = 1

# WHILE петља - извршава се све док је услов испуњен (i <= 5).
# Ова петља ће се извршити тачно 5 пута, једном за сваку вредност i од 1 до 5.
while i <= 5:
    # Исписивање реченице "Поздрав свима!" приликом сваког проласка кроз петљу.
    # Ова линија ће се извршити тачно 5 пута.
    print("Поздрав свима!")

    # Повећавање вредности бројача за 1 након сваког исписивања.
    # Без ове линије, вредност i би остала 1 и петља би се извршавала бесконачно.
    i = i + 1
    # i += 1  # Алтернативни начин за повећавање бројача за 1.

# Након што i постане 6, услов i <= 5 више није испуњен и петља се завршава.
# У овом тренутку, реченица "Поздрав свима!" је исписана тачно 5 пута.

```
