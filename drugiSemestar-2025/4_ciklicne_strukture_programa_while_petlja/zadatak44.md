```python
# Zadatak 44
# Програм рачуна суму бесконачног реда 1/1² + 1/2² + 1/3² + 1/4² + ... + 1/n² + ...
# док год је текући члан реда већи од 0.0001

# Иницијализација променљивих
i = 1  # бројач (индекс члана реда)
clan = 1  # променљива која чува вредност текућег члана, почиње од 1/1²
s = 0  # почетна вредност суме

# Петља која рачуна суму док је текући члан већи од 0.0001 (1e-4)
# 1e-4 се може написати и као 0.0001 или 1/10000
while clan > 1e-4:  # провера услова за наставак петље (1e-4 = 0.0001 - научна нотација)
    s = s + clan  # додај текући члан на суму
    i = i + 1  # повећај бројач (пређи на следећи члан)
    clan = 1 / (i * i)  # израчунај нови текући члан реда по формули 1/i²

# Исписивање резултата
# round(s, 3) заокружује суму на 3 децимале
print("Сума =", round(s, 3))  # исписује коначну суму реда

# Додавање математичке библиотеке
import math

# Иницијализација променљивих
i = 1  # бројач (индекс члана реда)
clan = 1  # променљива која чува вредност текућег члана, почиње од 1/1²
s = 0  # почетна вредност суме

# Петља која рачуна суму док је текући члан већи од 0.0001 (10^-4)
# math.pow(10, -4) користи математичку функцију за рачунање 10 на -4 (0.0001)
while clan > math.pow(10, -4):  # провера услова за наставак петље
    s = s + clan  # додај текући члан на суму
    i = i + 1  # повећај бројач (пређи на следећи члан)
    clan = 1 / (i * i)  # израчунај нови текући члан реда по формули 1/i²

# Исписивање резултата
# round(s, 3) заокружује суму на 3 децимале
print("Сума =", round(s, 3))  # исписује коначну суму реда
```
