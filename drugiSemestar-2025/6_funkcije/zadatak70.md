# Zadatak 70
# Саставити рекурзивну функцију која врши сабирање првих n бројева,
# а затим тестирати функцију за дато n и исписати добијени резултат.
# лавни део програма обрађује произвољан број уноса


def suma(n):
    if n == 0:
        return 0
    return n + suma(n - 1)


while True:
    n = int(input("n = "))
    s = suma(n)

    print(f"s {s}")
    print("----------")
