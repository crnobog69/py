```python
# Zadatak 66
# Програм рачуна суму алтернирајућег реда: 1/1!! - 1/2!! + 1/3!! - 1/4!! + ... + (-1)^(n+1)/n!!

# Функција за рачунање дуплог факторијала
# Дупли факторијел n!! је производ свих бројева од n до 1 који имају исту парност као n
# Пример: 5!! = 5 * 3 * 1 = 15, 6!! = 6 * 4 * 2 = 48
def dfakt(n):
    f = 1  # Почетна вредност производа
    while n >= 2:  # Док не дођемо до краја
        f = f * n  # Множимо тренутни производ са n
        n = n - 2  # Смањујемо n за 2 да бисмо прескочили број супротне парности
    return f


# Учитавање броја чланова реда
n = int(input("n = "))

# Иницијализација суме и знака
s = 0  # Почетна вредност суме
znak = 1  # Први сабирак је позитиван

# Рачунање суме алтернирајућег реда
for i in range(1, n + 1):
    s = s + znak * 1 / dfakt(i)  # Додавање i-тог члана реда на суму
    znak = -znak  # Промена знака за следећи члан

# Испис резултата заокруженог на 4 децимале
print(f"S = {round(s, 4)}")
```
