```python
# Zadatak 75

# Учитавање дужине низа од корисника
# Параметри: Нема
# Повратна вредност: Цео број који представља дужину низа
n = int(input("Дужина низа: n = "))

# Приказ поруке кориснику да унесе елементе низа
print("Унесите елементе низа: ")

# Иницијализација празне листе за чување елемената низа
niz = list()

# Иницијализација променљиве за суму елемената, почетна вредност је 0
s = 0

# Блок за унос елемената низа
# Петља се извршава n пута, од 0 до n-1
for i in range(0, n):
    # Учитавање појединачног елемента низа од корисника
    # Параметри: Нема
    # Повратна вредност: Цео број који представља елемент низа
    temp = int(input())

    # Додавање елемента у низ помоћу методе append()
    # Параметри: temp - вредност која се додаје у низ
    # Повратна вредност: Нема (модификује постојећи низ)
    niz.append(temp)

# Блок за рачунање суме свих елемената низа
for i in range(0, n):
    # Додавање вредности елемента на позицији i суми s
    s = s + niz[i]

# Рачунање аритметичке средине и заокруживање на 5 децимала
# Параметри: Израз s/n - аритметичка средина, 5 - број децимала
# Повратна вредност: Број заокружен на 5 децимала
asr = round(s / n, 5)

# Приказ израчунате аритметичке средине кориснику
# Користи се f-string за форматирање излаза
print(f"Аритметичка средина је {asr}")

```
