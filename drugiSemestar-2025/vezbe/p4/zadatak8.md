```python
# Zadatak 70

# Рачунање суме бројева од 1 до n рекурзивно
# Рекурзивна функција која израчунава суму бројева од 1 до n
def suma(n):
    # Базни случај: ако је n једнако 0, сума је 0
    if n == 0:
        return 0
    # Рекурзивни случај: додај n на суму свих бројева до n-1
    return n + suma(n - 1)


# Главна петља програма која се извршава неограничено
while True:
    # Унос броја n од корисника
    n = int(input("n = "))
    # Позивање функције за израчунавање суме и памћење резултата
    s = suma(n)

    # Приказ резултата суме
    print(f"s {s}")
    # Приказ раздвајача за боље читање резултата
    print("-------")

```
