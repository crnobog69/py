```python
# Zadatak 65


def fakt(x):
    # Функција за рачунање факторијела броја x.
    # Формула: x! = 1 * 2 * 3 * ... * x

    f = 1  # Почетна вредност факторијела је 1
    for i in range(1, x + 1):  # Петља од 1 до x (укључујући x)
        f = f * i  # Множимо тренутну вредност са бројем i
    return f  # Враћамо израчунати факторијел


# Унос вредности од корисника
n = int(input("n = "))  # Унос броја n
k = int(input("k = "))  # Унос броја k

# Рачунање биномног коефицијента C(n,k) по формули n!/(k!(n-k)!)
c = fakt(n) // (fakt(k) * fakt(n - k))

# Приказ резултата
print(f"c = {c}")  # Испис израчунатог биномног коефицијента

```
