```python
# Zadatak 63

# Функција која проверава да ли је број паран или непаран
def paran(n):
    # Провера парности броја помоћу операције модуло
    # Ако је остатак при дељењу са 2 једнак 0, број је паран
    if n % 2 == 0:
        print("Број је паран.")
    else:
        print("Број је непаран.")


# Функција која проверава да ли је број позитиван, негативан или нула
def negativan(n):
    # Провера да ли је број мањи од нуле
    if n < 0:
        print("Број је негативан")
    # Провера да ли је број већи од нуле
    elif n > 0:
        print("Број је позитиван.")
    # Ако број није ни мањи ни већи од нуле, онда је једнак нули
    else:
        print("Број је нула.")


# Учитавање целобројне вредности од корисника
x = int(input("x = "))

# Позив функције за проверу парности броја
paran(x)
# Позив функције за проверу знака броја
negativan(x)

```
