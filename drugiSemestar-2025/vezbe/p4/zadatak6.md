```python
# Zadatak 67
# Програм за игру погађања бројева у којој рачунар замишља број, а играч покушава да га погоди

import random


def partija(n, m):
    """
    Функција која симулира једну партију игре погађања бројева.

    Параметри:
    n - горња граница опсега из којег се број бира (од 1 до n)
    m - број покушаја након којег рачунар мења замишљени број

    Повратна вредност:
    -1 - ако је играч унео неисправан број (ван опсега)
    број покушаја - ако је играч погодио број
    """
    pokusaji = 0  # Бројач покушаја играча

    x = random.randint(1, n + 1)  # Рачунар замишља случајан број из опсега [1, n+1)

    while True:
        p = int(input("Ваш број: "))  # Играч уноси свој покушај
        pokusaji += 1  # Повећава се бројач покушаја

        if p < 1 or p > n:  # Провера да ли је унешени број у дозвољеном опсегу
            return -1  # Ако није у опсегу, игра се прекида
        elif p == x:  # Провера да ли је играч погодио број
            return pokusaji  # Враћа број покушаја који је био потребан да се погоди
        elif pokusaji % m == 0:  # Провера да ли треба променити замишљени број
            print(f"Мењам број! Замишљени број је - {x}")  # Обавештење о промени броја
            x = random.randint(1, n + 1)  # Рачунар замишља нови случајан број


# Главна петља програма која омогућава више партија
while True:
    print("---------------Нова Игра----------------")  # Исписује назив нове игре

    n = int(input("Опсег: n = "))  # Унос горње границе опсега
    m = int(
        input("Број покушаја: m(m<=n) = ")
    )  # Унос броја покушаја након којег се мења број

    # Провера да ли су унети параметри исправни
    if n > 1 and m >= 1 and m <= n:
        rezultat = partija(n, m)  # Покреће се партија са задатим параметрима

        if rezultat == -1:  # Ако је враћено -1, играч је унео неисправан број
            print("Било је занимљиво играти са вама!")  # Порука о крају игре
            break  # Прекид главне петље - крај програма
        else:
            # Обавештење о успешном погађању
            print(
                f"Погодили сте из {rezultat} пута!"
            )  # Приказује број покушаја који је био потребан
    # Ако параметри нису исправни, програм тражи нове параметре (не исписује грешку)

```
