```python
# Zadatak 42
# Програм рачуна суму: S = 1/(3²) + 1/(5²) + 1/(7²) + ... + 1/((2n+1)²)

n = int(input(" n = "))  # Учитавање броја n од корисника

i = 1  # Бројач петље
s = 0  # Променљива за чување резултата суме

while i <= n:  # Петља се извршава n пута
    s = s + 1 / ((2 * i + 1) * (2 * i + 1))  # Додајемо вредност 1/((2i+1)²) суми
    i = i + 1  # Повећавамо бројач за 1
print(f"S = {round(s, 5)}")  # Исписујемо резултат заокружен на 5 децимала
```
