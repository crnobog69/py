# Zadatak 48
# Програм исписује троугао бројева где сваки ред садржи
# парне бројеве до двоструке вредности редног броја колоне

# Учитавање броја редова од корисника
n = int(input("Broj redova = "))

# Иницијализација бројача редова
i = 1

# Спољашња петља која контролише број редова
while i <= n:
    # Иницијализација бројача колона за сваки ред
    j = 1
    # Унутрашња петља која контролише број колона у текућем реду
    while j <= i:
        # Провера да ли је колона у оквиру тренутног реда
        if i >= j:
            # Исписивање двоструке вредности бројача колоне
            # Параметар end="" спречава прелазак у нови ред након испис вредности и
            # уместо подразумеваног краја реда (\n) користи празан стринг
            # Ово омогућава да се сви бројеви у истом реду испишу један поред другог
            print(2 * j, " ", end="")
        # Инкрементирање бројача колона
        j = j + 1
    # Прелазак у нови ред након завршетка текућег реда
    print()
    # Инкрементирање бројача редова
    i = i + 1
