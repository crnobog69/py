```python
# Zadatak 40-a
# Програм приказује првих н Фибоначијевих бројева

broj_clanova = int(
    input(" n = ")
)  # Учитавамо број н са тастатуре (број Фибоначијевих бројева)

redni_broj = 3  # Бројач који почиње од 3 (јер су прва два броја већ дефинисана)
fib_prvi = 1  # Први члан низа (f₁)
fib_drugi = 1  # Други члан низа (f₂)

print("FIBONACIJEVI BROJEVI")
print("f1 = 1\nf2 = 1")  # Исписујемо прва два Фибоначијева броја

# Петља која рачуна Фибоначијеве бројеве од трећег до н-тог
while redni_broj <= broj_clanova:
    fib_trenutni = (
        fib_prvi + fib_drugi
    )  # Израчунавамо следећи Фибоначијев број као збир два претходна
    fib_prви = fib_drugi  # Први члан постаје други
    fib_drugi = fib_trenutni  # Други члан постаје тренутни (новоизрачунати)
    print("f", redni_broj, sep="", end="")  # Исписујемо индекс Фибоначијевог броја
    print(" =", fib_trenutni)  # Исписујемо вредност Фибоначијевог броја
    redni_broj = redni_broj + 1  # Повећавамо бројач
```
