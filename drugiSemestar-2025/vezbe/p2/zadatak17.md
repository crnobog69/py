```python
# Задатак 36

# Учитавање целобројног уноса од корисника
n = int(input("n = "))

print()

# Исписивање горњег дела шаблона
# Овај део кода исписује редове од n до 1
for i in range(n, 0, -1):
    # Исписивање водећих размака
    # Овај део кода исписује размаке пре бројева у сваком реду
    for k in range(n, i, -1):
        print(" ", end="")
    # Исписивање бројева
    # Овај део кода исписује бројеве у сваком реду
    for j in range(1, i + 1, 1):
        print(i, "", end="")
    print()

# Исписивање доњег дела шаблона
# Овај део кода исписује редове од 2 до n
for i in range(2, n + 1, 1):
    # Исписивање водећих размака
    # Овај део кода исписује размаке пре бројева у сваком реду
    for k in range(n, i, -1):
        print(" ", end="")
    # Исписивање бројева
    # Овај део кода исписује бројеве у сваком реду
    for j in range(1, i + 1, 1):
        print(i, "", end="")
    print()
```
