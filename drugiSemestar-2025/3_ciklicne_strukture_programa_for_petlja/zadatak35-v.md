```python
# Zadatak 35-v
# Програм за исцртавање троугла од звездица (*) у облику:
# n = 5
# *
# * *
# * * *
# * * * *
# * * * * *
# ... (до н редова)

# Учитавање броја н од корисника - овај број одређује висину троугла
n = int(input("n = "))

# Спољна петлја која пролази кроз сваки ред троугла
# i иде од 1 до n (укључујући n)
for i in range(1, n + 1, 1):
    # Унутрашња петља која контролише број звездица у тренутном реду
    # У сваком реду i имамо тачно i звездица
    for j in range(1, i + 1, 1):
        # Исписујемо звездицу и размак без преласка у нови ред (end="")
        print("* ", end="")
    # Након што завршимо са исписивањем звездица у једном реду,
    # прелазимо у нови ред за следећи ред троугла
    print()

```
