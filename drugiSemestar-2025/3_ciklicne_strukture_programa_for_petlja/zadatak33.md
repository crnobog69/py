```python
# Zadatak 33
# Саставити програм који ће исписати све делиоце унетог броја n и обавештење да ли је унети број савршен.
# Број је савршен ако је једнак суми својих делиоца искључујући њега самог.
# На пример, 28=1+2+4+7+14.

# Учитавамо цео број n са тастатуре
n = int(input("Унесите n = "))

# Иницијализујемо променљиву s која ће чувати суму свих делилаца броја n
s = 0

# Исписујемо наслов пре набрајања делилаца
print("Делиоци: ")

# Пролазимо кроз све бројеве од 1 до n (укључујући оба)
for i in range(1, n + 1, 1):
    # Проверавамо да ли је i делилац броја n (да ли n дељено са i даје остатак 0)
    if n % i == 0:
        # Исписујемо пронађени делилац
        print(i)
        # Додајемо делилац у укупну суму делилаца
        s = s + i

# Проверавамо да ли је број савршен
# Савршен број је једнак суми својих делилаца без себе самог
# Од суме свих делилаца (s) одузимамо сам број (n) и проверавамо да ли је једнак броју n
if s - n == n:
    # Ако је услов испуњен, број је савршен
    print(f"Број {n} јесте саврштен.")
else:
    # Ако услов није испуњен, број није савршен
    print(f"Број {n} није савршен.")

# =============================================
# Оптимизована верзија кода (алтернативно решење)
# =============================================

print("\n\n# Оптимизована верзија решења:")
print("# ============================")

# Учитавамо цео број за оптимизовану верзију
n = int(input("Унесите број за проверу n = "))

# Иницијализујемо променљиву за суму правих делилаца (без самог броја)
suma_delilaca = 0

print("Делиоци: ")

# Пролазимо само до n//2 јер већи делиоци не постоје (осим самог n)
for i in range(1, n // 2 + 1):
    if n % i == 0:
        print(i)
        suma_delilaca += i

# n је увек делилац самог себе
print(n)

# Једноставнија провера савршеног броја - поредимо директно са сумом правих делилаца
if suma_delilaca == n:
    print(f"Број {n} јесте савршен.")
else:
    print(f"Број {n} није савршен.")
```
