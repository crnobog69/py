# Zadatak 31
# Саставити програм којим се исписују сви троцифрени Амстронгови бројеви.
# Троцифрени број је Амстронгов ако је једнак збиру кубова својих цифара

# Програм за налажење свих троцифрених Армстронгових бројева
# Армстронгов број је број који је једнак збиру степена својих цифара
# У овом задатку тражимо троцифрене бројеве где је број једнак збиру кубова својих цифара

print("Армстронгови бројеви: ")

# Користимо три угнежђене петље да генеришемо све троцифрене бројеве
# Прва петља (a) генерише прву цифру троцифреног броја (стотине)
for a in range(1, 9 + 1, 1):
    # Друга петља (b) генерише другу цифру троцифреног броја (десетице)
    for b in range(0, 9 + 1, 1):
        # Трећа петља (c) генерише трећу цифру троцифреног броја (јединице)
        for c in range(0, 9 + 1, 1):
            # Формирамо троцифрени број: а*100 + b*10 + c
            # Нпр. ако је a=1, b=5, c=3, број је 153
            broj = 100 * a + 10 * b + c

            # Проверавамо да ли је број Армстронгов
            # Рачунамо збир кубова цифара: a³ + b³ + c³
            # И упоређујемо са оригиналним бројем
            if broj == a * a * a + b * b * b + c * c * c:
                # Ако је услов испуњен, исписујемо број
                print(broj)
                # Пример: 153 = 1³ + 5³ + 3³ = 1 + 125 + 27 = 153

# ---------- ИЛИ ----------

# Користимо једну петљу која пролази кроз све троцифрене бројеве (100-999)
for broj in range(100, 1000):
    # Претварамо број у стринг, затим креирамо листу цифара, и подижемо сваку на куб
    # Користећи листну компрехензију
    cifre = [int(cifra) for cifra in str(broj)]

    # Рачунамо збир кубова цифара користећи листу
    zbir_kubova = sum(cifra**3 for cifra in cifre)

    # Проверавамо да ли је број Армстронгов
    if broj == zbir_kubova:
        print(broj)

# ---------- ИЛИ ----------

# Користимо једну петљу која пролази кроз све троцифрене бројеве (100-999)
for broj in range(100, 1000):
    # Издвајамо цифре броја
    stotine = broj // 100  # Прва цифра (стотине)
    desetice = (broj // 10) % 10  # Друга цифра (десетице)
    jedinice = broj % 10  # Трећа цифра (јединице)

    # Рачунамо збир кубова цифара
    zbir_kubova = stotine**3 + desetice**3 + jedinice**3

    # Проверавамо да ли је број Армстронгов
    if broj == zbir_kubova:
        print(broj)
