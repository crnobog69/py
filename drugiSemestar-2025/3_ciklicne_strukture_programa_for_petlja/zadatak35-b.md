```python
# Задатак 35-б
# Програм који исцртава шупљи квадрат од звездица димензија n×n

# n = 5
# * * * * *
# *       *
# *       *
# *       *
# * * * * *

# Учитавамо број n који одређује димензије квадрата
n = int(input("n = "))

# Спољашња петља - итерира кроз редове квадрата
# i представља тренутни ред (од 1 до n)
for i in range(1, n + 1, 1):
    # Унутрашња петља - итерира кроз колоне у сваком реду
    # j представља тренутну колону (од 1 до n)
    for j in range(1, n + 1, 1):
        # Проверавамо да ли смо у првом или последњем реду
        if i == 1 or i == n:
            # Ако јесмо, исписујемо звездицу за сваку позицију у реду
            print("* ", end="")
        # Ако нисмо у првом или последњем реду, проверавамо да ли смо у првој или последњој колони
        elif j == 1 or j == n:
            # Ако јесмо, исписујемо звездицу (ово су ивице квадрата)
            print("* ", end="")
        # Ако нисмо ни на једној ивици квадрата, налазимо се унутар квадрата
        else:
            # Исписујемо размаке да бисмо добили шупљи квадрат
            print("  ", end="")
    # Након сваког реда прелазимо у нови ред користећи празан print()
    print()
# Када се програм изврши, резултат је шупљи квадрат од звездица димензија n×n

```
