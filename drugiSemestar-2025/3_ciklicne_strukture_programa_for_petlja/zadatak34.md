```python
# Zadatak 33
# Саставити програм за табелирање функције y = 2x + 1 / x^2 - 1 у опсегу од xmin до xmax са кораком dx.

# Програм табелира вредности функције y = (2x + 1) / (x^2 - 1) за опсег вредности x
# Напомена: функција има сингуларитете (није дефинисана) за x = 1 и x = -1

# Учитавање граница интервала и корака
xmin = int(input("xmin = "))  # Доња граница интервала за x
xmax = int(input("xmax = "))  # Горња граница интервала за x (искључена из опсега)
dx = int(input("dx = "))  # Корак (инкремент) за x

# Исписивање заглавља табеле
print("     x       y")  # Заглавље колона, формат са размацима за поравнање

# Итерација кроз вредности x од xmin до xmax-1 са кораком dx
for x in range(xmin, xmax, dx):
    # Израчунавање вредности функције за тренутно x
    # Пажња: ово ће изазвати грешку дељења са нулом ако је x = 1 или x = -1
    y = (2 * x + 1) / (x * x - 1)

    # Форматирање и приказ вредности x као целог броја са ширином 5 места
    print(format(x, "5d"), end="")

    # Форматирање и приказ вредности y као децималног броја
    # Заокруживање на 2 децимале, приказ на 8 места укључујући децималну тачку
    print(format(round(y, 2), "8.2f"))

```
