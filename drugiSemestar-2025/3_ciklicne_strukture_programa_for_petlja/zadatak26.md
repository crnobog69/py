```python
# Zadatak 26

# Саставити програм који за унето n исписује збир првих n целих позитивних бројева.

# Програм рачуна суму првих n природних бројева
# Формула: 1 + 2 + 3 + ... + n

# Учитавање броја n са тастатуре и конверзија у цео број
n = int(input(" Унесите n = "))

# Иницијализација променљиве за суму
s = 0

# Петља која пролази кроз све бројеве од 1 до n
# range(1, n+1, 1) генерише бројеве: 1, 2, 3, ..., n
for i in range(1, n + 1, 1):
    # Додавање тренутног броја i у суму s
    s = s + i

# Исписивање резултата - укупне суме свих бројева од 1 до n
print(f" Suma = {s}")

# Напомена: Ова сума се може израчунати и формулом: n*(n+1)/2
# Објашњење формуле:
# Ако напишемо суму два пута, једном од почетка, а једном од краја:
# S = 1 + 2 + 3 + ... + (n-1) + n
# S = n + (n-1) + (n-2) + ... + 2 + 1
# Сабирањем вертикално добијамо:
# 2S = (1+n) + (2+(n-1)) + (3+(n-2)) + ... + ((n-1)+2) + (n+1)
# 2S = (n+1) + (n+1) + (n+1) + ... + (n+1) + (n+1) [имамо n парова]
# 2S = n*(n+1)
# S = n*(n+1)/2
# Ово је Гаусова формула за суму аритметичке прогресије.
```
