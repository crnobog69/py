```python
# Zadatak 35-đ

# n = 5
# * * * * *
# *     *
# *   *
# * *
# *

# Учитавамо број n који одређује величину обрасца
n = int(input("n = "))

# Спољашња петља контролише редове (од 1 до n)
for i in range(1, n + 1, 1):
    # Унутрашња петља контролише колоне
    # Крећемо од n и идемо до позиције i (укључујући i)
    for j in range(n, i - 1, -1):
        # Ако смо у првом реду, исцртавамо све звездице
        if i == 1:
            print("* ", end="")
        # Иначе, звездице исцртавамо само на левој ивици (j == n) и дијагонали (j == i)
        elif j == n or j == i:
            print("* ", end="")
        # На свим осталим позицијама исцртавамо размаке
        else:
            print("  ", end="")
    # Прелазимо у нови ред након што исцртамо све колоне за тренутни ред
    print()

```
