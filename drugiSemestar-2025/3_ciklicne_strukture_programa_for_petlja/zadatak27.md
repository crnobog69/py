```python
# Zadatak 27
# Саставити програм којим се за унети природан број n израчувава суму:
# S = 1! + 2! + 3! + ... + n!

# Улаз: корисник уноси природан број n
n = int(input("Унесите n = "))

# Иницијализација променљивих:
# s - променљива за чување суме факторијела, почиње од 0
# f - променљива за чување вредности факторијела тренутног броја, почиње од 1
s = 0
f = 1

# Петља која израчунава факторијеле бројева од 1 до n и додаје их у суму
for i in range(1, n + 1, 1):
    # Израчунавање факторијела броја i
    # f = f * i је ефикасан начин рачунања јер користи претходни резултат
    # На пример, када i=3, f већ садржи 2!, па множењем са 3 добијамо 3!
    f = f * i

    # Додавање израчунатог факторијела у суму
    s = s + f

# Исписивање коначног резултата
print(f"Сума = {s}")

```
