```python
# Zadatak 29
# Саставити програм којим се, за дати природан број n и реалан број x,
# израчунава израз:
# S = cos(x) + cos(x^2) + ... + cos(x^n)

# Увоз математичке библиотеке која садржи тригонометријске функције
import math

# Учитавање улазних података од корисника
# n - природан број који представља горњу границу степена
n = int(input("Унесите n = "))
# x - реалан број који се степенује
# Напомена: требало би користити float уместо int за x пошто је по задатку реалан број
x = float(input("Унесите x = "))

# Иницијализација променљивих
# suma - променљива у којој се чува сума косинуса
suma = 0
# stepen_x - променљива у којој се чува тренутна вредност степена x-а
stepen_x = 1

# Циклус који израчунава суму косинуса
# Итерира од 1 до n (укључујући n)
for i in range(1, n + 1, 1):
    # Рачунање следећег степена броја x множењем stepen_x са x
    # У свакој итерацији stepen_x постаје: x, x^2, x^3, ..., x^n
    stepen_x = stepen_x * x

    # Додавање косинуса тренутног степена у суму
    # У свакој итерацији додајемо: cos(x), cos(x^2), cos(x^3), ..., cos(x^n)
    suma = suma + math.cos(stepen_x)

# Испис резултата са форматирањем на 5 децимала
# Користи се f-string за лепши и прегледнији испис
print(f"Сума = {suma:.5f}")

# Увоз математичке библиотеке која садржи тригонометријске функције
# import math

# Учитавање улазних података од корисника
n = int(input("Унесите n = "))
x = float(input("Унесите x = "))

# Једноставно израчунавање суме помоћу листе и оператора степеновања (**)
# x**i директно рачуна x на степен i
# Ово елиминише потребу за променљивом која памти тренутни степен
suma = sum(math.cos(math.pow(x, i)) for i in range(1, n + 1))

# Испис резултата са форматирањем на 5 децимала
print(f"Сума = {suma:.5f}")
```
