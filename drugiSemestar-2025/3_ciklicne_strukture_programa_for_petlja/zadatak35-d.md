```python
# Zadatak 35-d
#
# Програм исцртава образац звездица у облику обрнутог троугла.
# За унети број n, програм црта n редова где сваки ред има
# све мањи број звездица, почевши од n звездица у првом реду.

# n = 5
# * * * * *
# * * * *
# * * *
# * *
# *

# Учитавамо број редова од корисника
n = int(input("n = "))

# Спољашња петља контролише број редова
# i узима вредности од 1 до n (укључујући n)
for i in range(1, n + 1, 1):
    # Унутрашња петља исцртава звездице у текућем реду
    # Број звездица у реду је (n - i + 1)
    # Петља иде од n уназад до i (укључујући i)
    for j in range(n, i - 1, -1):
        # Исписујемо звездицу праћену размаком, без преласка у нови ред
        print("* ", end="")
    # Након исцртавања свих звездица у реду, прелазимо у нови ред
    print()

```
