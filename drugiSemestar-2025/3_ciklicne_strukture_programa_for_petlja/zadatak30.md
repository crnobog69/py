```python
# Zadatak 30

# Саставити програм који ће исписати све бројева у опсегу од a до b,
# који су дељиви са бројем 7 и њихову суму

# Учитавање граница интервала од корисника
a = int(input("Унесите a = "))  # Доња граница интервала
b = int(input("Унесите b = "))  # Горња граница интервала

# Иницијализација променљиве за чување суме бројева дељивих са 7
s = 0

# Испис поруке која објашњава шта следи у излазу
print("Бројеви од", a, "до", b, "дељиви са 7:")

# Пролазак кроз све бројеве у интервалу [a, b]
for i in range(a, b + 1, 1):
    # Проверавамо да ли је тренутни број дељив са 7
    if i % 7 == 0:
        s = s + i  # Додајемо број у суму ако је дељив са 7
        print(i)  # Исписујемо број који је дељив са 7

# Испис коначне суме свих бројева дељивих са 7 у задатом опсегу
print(f"Сума = {s}")

```
