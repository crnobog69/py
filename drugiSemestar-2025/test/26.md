```python
# 26

# Програм за валидацију корисничког имена
# Проверава дужину, присуство размака и бројева

# input() - функција која прима текстуални унос од корисника и враћа га као string
korisnickoIme = input("Унесите ваше корисничко име: ")

# len() - функција која враћа број карактера у стрингу
# Проверава се да ли је дужина корисничког имена већа од 12
if len(korisnickoIme) > 12:
    print("Корисничко не може да садржи више од 12 слова.")
# find() - функција која тражи подстринг у стрингу и враћа позицију првог појављивања или -1 ако не постоји
# Проверава се да ли корисничко име садржи размак
elif not korisnickoIme.find(" ") == -1:
    print("Корисничко име не море да садржи размаке.")
# isalpha() - функција која враћа True ако се стринг састоји само од слова, иначе False
# Проверава се да ли корисничко име садржи само слова
elif not korisnickoIme.isalpha():
    print("Корисничко име не може да садржи бројеве.")
# Ако су све провере успешне, користи се f-string за форматирање поруке
# f-string - омогућава уметање променљивих директно у стринг помоћу синтаксе {променљива}
else:
    print(f"Здраво, {korisnickoIme}!")

```
