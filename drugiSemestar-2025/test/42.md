```python
# 42

## Квиз о Србији - програм који тестира знање о географији и историји Србије

# Листа свих питања у квизу
pitanja = (
    "Које је главни град Србије?",
    "Које је највеће језеро у Србији?",
    "Која планина у Србији је највиша?",
    "Која река је најдужа у Србији?",
    "Који научник је рођен у Смиљану?",
)

# Понуђени одговори за свако питање
opcije = (
    ("1. Београд", "2. Нови Сад", "3. Ниш", "4. Крагујевац"),
    ("1. Ђердапско", "2. Власинско", "3. Палићко", "4. Перућачко"),
    ("1. Стара планина", "2. Копаоник", "3. Златибор", "4. Шар-планина"),
    ("1. Дунав", "2. Сава", "3. Морава", "4. Дрина"),
    ("1. Никола Тесла", "2. Михајло Пупин", "3. Милутин Миланковић", "4. Јосиф Панчић"),
)

# Тачни одговори - бројеви од 1 до 4
odgovori = ("1", "1", "4", "1", "1")
odgovorKorisnika = []  # Листа за чување корисничких одговора
rezultat = 0  # Бројач тачних одговора
brojPitanja = 0  # Индекс тренутног питања

# Пролазак кроз сва питања у квизу
for pitanje in pitanja:
    print("-" * 20)  # Линија раздвајања за бољу прегледност
    print(pitanje)

    # Приказ свих понуђених одговора за тренутно питање
    for opcija in opcije[brojPitanja]:
        print(opcija)

    # Унос одговора од стране корисника
    odgovor = input("Ваш одговор (1, 2, 3, 4): ")
    odgovorKorisnika.append(odgovor)  # Чување одговора

    # Провера да ли је одговор тачан
    if odgovor == odgovori[brojPitanja]:
        rezultat += 1
        print()
        print("Тачно!")
        print()
    else:
        print()
        print("Нетачно!")
        print()
        print(f"{odgovori[brojPitanja]} је тачан одговор.")
        print()

    brojPitanja += 1  # Повећање индекса за следеће питање

rezultatiTekst = "РЕЗУЛТАТИ"

print("-" * 20)
print(f"{rezultatiTekst:^21}")
print("-" * 20)

print("Одгопвори: ", end=" ")
for _ in odgovori:
    print(_, end=" ")
print()

print("Одговори корисника: ", end=" ")
for _ in odgovorKorisnika:
    print(_, end=" ")
print()

rezultat = int(rezultat / len(pitanja) * 100)
print(f"Ваш резултат је: {rezultat}%")
```
