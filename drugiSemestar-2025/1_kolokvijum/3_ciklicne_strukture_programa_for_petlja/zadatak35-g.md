```python
# Zadatak 35-g

# n = 5
# *
# * *
# *   *
# *     *
# * * * * *

# Унос броја редова од корисника
n = int(input("n = "))

# Спољашња петља - контролише број редова
# Почиње од 1 и иде до н (укључујући н)
for i in range(1, n + 1, 1):
    # Унутрашња петља - контролише број колона у сваком реду
    # За сваки ред и, имамо и колона
    for j in range(1, i + 1, 1):
        # Ако је последњи ред (и == н), исписујемо све звездице
        if i == n:
            print("* ", end="")
        # Ако није последњи ред, али јесте прва или последња позиција у реду
        # исписујемо звездицу да формирамо ивице троугла
        elif j == 1 or j == i:
            print("* ", end="")
        # За све остале позиције исписујемо празнине
        # тако да се види само контура троугла
        else:
            print("  ", end="")
    # Прелазак у нови ред након завршетка сваког реда шаблона
    print()

```
