```python
# Zadatak 35-z

# n = 5
# * * * * *
#  * * * *
#   * * *
#    * *
#     *

# Учитавање броја редова троугла са тастатуре
n = int(input("n = "))

# Спољна петља која контролише број редова
# Итерација иде од n до 1 (у опадајућем редоследу)
for i in range(n, 0, -1):
    # Прва унутрашња петља исцртава размаке на почетку сваког реда
    # Број размака је једнак тренутном индексу реда (n-i)
    for k in range(n, i, -1):
        print(" ", end="")

    # Друга унутрашња петља исцртава звездице и размаке
    # Број звездица у сваком реду је једнак вредности i
    for j in range(1, i + 1, 1):
        print("* ", end="")

    # Прелазак у нови ред након исцртавања једног реда троугла
    print()

```
