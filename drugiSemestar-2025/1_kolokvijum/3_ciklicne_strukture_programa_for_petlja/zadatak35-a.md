```python
# Zadatak 25-a

# n=5:
# * * * * *
# * * * * *
# * * * * *
# * * * * *
# * * * * *

# Учитавање броја 'n' од корисника који одређује величину матрице (број редова и колона)
n = int(input("n = "))

# Спољашња петлја - креирање редова матрице
# Променљива 'i' представља тренутни ред (од 1 до n)
for i in range(1, n + 1, 1):
    # Унутрашња петља - креиранје колона унутар тренутног реда
    # Променљива 'j' представља тренутну колону (од 1 до n)
    for j in range(1, n + 1, 1):
        # Штампање звезде са размаком без преласка у нови ред (end="")
        # Ово поставља звездице једну поред друге у истом реду
        print("*  ", end="")

    # Након што се исцрта цео ред (n звездица), прелазимо у нови ред
    # Ова наредба print() без параметара само помера курсор у нови ред
    print()
```
