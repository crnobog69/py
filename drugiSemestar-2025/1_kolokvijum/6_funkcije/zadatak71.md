```python
# Zadatak 71
# Саставити рекурзивну функцију за одређивање највећег заједничког делилаца за два цела броја.
# Затим тестирати функцију у главном програму и исписати добијени резултат.
# Главни део програма обрађује произвољан број уноса.


# Рекурзивна функција за одређивање највећег заједничког делиоца (НЗД)
# Параметри:
#   x - први број (већи од два броја)
#   y - други број (мањи од два броја)
# Повратна вредност:
#   цео број који представља НЗД бројева x и y
def NZD(x, y):
    # Базни случај рекурзије: ако је y делилац броја x
    # тада је y највећи заједнички делилац за x и y
    if x % y == 0:
        return y

    # Рекурзивни позив: примена Еуклидовог алгоритма
    # НЗД(x, y) = НЗД(y, x % y), где је x % y остатак дељења x са y
    return NZD(y, x % y)


# Главни део програма - бесконачна петља за обраду произвољног броја уноса
while True:
    # Унос првог броја од корисника
    n = int(input("n = "))

    # Унос другог броја од корисника
    m = int(input("m = "))

    # Провера да ли је први број мањи од другог
    # Ако јесте, замени њихове вредности
    # Ово осигурава да функцији NZD увек прослеђујемо већи број као први параметар
    if n < m:
        n, m = m, n  # Истовремена замена вредности променљивих

    # Испис резултата - позив функције NZD са бројевима n и m
    print(f"НЗД = {NZD(n, m)}")

    # Испис линије која одваја различите уносе
    print("----------")

```
