```python
# Zadatak 69
# Саставити рекурзивну функцију која врши степеновање целог броја на целобројни изложилац,
# а затим тестирати функцију за дати природни број и изложилац и исписати добијени резултат.
# Главни део програма обрађује произвољан број

# Дефиниција рекурзивне функције стepen која рачуна степен броја
# @param a - основа степена (број који се степенује)
# @param k - изложилац (степен на који подижемо број)
# @return - враћа број a подигнут на степен k (a^k)
def stepen(a, k):
    # Базни случај рекурзије - било који број на степен 0 је једнако 1
    if k == 0:
        return 1
    # Рекурзивни корак - множимо a са резултатом рекурзивног позива функције
    # за степен који је умањен за 1 (a * a^(k-1) = a^k)
    # Изложилац се умањује за 1 у сваком кораку рекурзије да бисмо се приближили базном случају.
    # Ово је у складу са математичком дефиницијом степеновања: a^k = a * a^(k-1)
    # Нпр. 2^3 = 2 * 2^2 = 2 * (2 * 2^1) = 2 * (2 * (2 * 2^0)) = 2 * (2 * (2 * 1)) = 8
    return a * stepen(a, k - 1)


# Главни део програма - бесконачна петља за тестирање функције
while True:
    # Учитавање основе степена од корисника
    a = int(input("a = "))
    # Учитавање изложиоца од корисника
    k = int(input("k = "))

    # Исписивање резултата позивом функције stepen
    print(f"степен = {stepen(a, k)}")
    # Исписивање раздвајача за бољу прегледност између тестова
    print("----------")

```
