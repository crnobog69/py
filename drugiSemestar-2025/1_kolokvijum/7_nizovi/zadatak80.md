```python
# Zadatak 80
# Саставити програм за сортирање унетог низа целих бројева,
# дужине n,
# у неопадајући поредак методом избора (Selection Sort). Исписати сортирани низ.
# Selection Sort подразумева да минимални елемент низа размени са a[0],
# минимални елемент одсечка a[1], a[2], …, a[n-1]  разменити са a[1],
# минимални елемент одсечка a[2], a[3], …, a[n-1]  разменити са a[2];
# исти поступак применити на преостале елементе осим последњег који се налази на свом месту.

# Креирамо празну листу која ће садржати елементе низа
niz = list()

# Унос броја елемената низа (величина низа)
# int() - функција која конвертује стринг у цео број
# input() - функција која чита улаз од корисника као стринг
n = int(input("Дужина низа: n = "))
print("унесите елементе низа: ")

# Петља која се извршава n пута за унос n елемената низа
# range(0, n) - генерише секвенцу бројева од 0 до n-1
# i - представља текући индекс у петљи
for i in range(0, n):
    # Читамо број са улаза и конвертујемо га у целобројну вредност
    temp = int(input())
    # Додајемо прочитани број на крај листе
    # append() - метода која додаје елемент на крај листе
    niz.append(temp)

# Имплементација Selection Sort алгоритма
# Спољашња петља иде од првог до претпоследњег елемента (индекси од 0 до n-2)
# Сваки пролаз кроз ову петљу поставља један елемент на коначну позицију
for i in range(0, n - 1):
    # Унутрашња петља пролази кроз несортирани део низа
    # Креће од елемента након тренутне позиције i (i+1) и иде до краја низа (n)
    for j in range(i + 1, n):
        # Поређење елемената - ако је елемент на позицији i већи од елемента на позицији j,
        # замени их местима (размена се врши да би се мањи елемент поставио на позицију i)
        if niz[i] > niz[j]:
            # Замена елемената користећи Python синтаксу за вишеструку доделу
            # Ова линија је еквивалентна следећем коду:
            # temp = niz[i]
            # niz[i] = niz[j]
            # niz[j] = temp
            niz[i], niz[j] = niz[j], niz[i]

# Исписивање сортираног низа
print("Сортирани низ: ")
# Исписује целу листу у једној линији
print(niz)
```
