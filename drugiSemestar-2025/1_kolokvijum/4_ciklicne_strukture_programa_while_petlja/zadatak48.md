```python
# Zadatak 48
# Програм који исцртава троугао са парним бројевима користећи while петље

# Учитавање броја редова од корисника
# Функција input() прихвата унос од корисника као стринг
# Функција int() конвертује стринг у цео број
# Променљива n означава укупан број редова троугла
n = int(input("Број редова = "))

# Иницијализација бројача редова
# i представља тренутни ред који се исцртава
i = 1

# Спољашња while петља - извршава се n пута (једном за сваки ред)
# Услов i <= n осигурава да се петља извршава тачно n пута
while i <= n:
    # Иницијализација бројача колона
    # j представља позицију унутар тренутног реда i
    j = 1

    # Унутрашња while петља - исцртава елементе у тренутном реду
    # Петља се извршава i пута за сваки ред (јер у i-том реду имамо i елемената)
    while j <= i:
        # Провера да ли треба исписати број
        # Услов i >= j је увек тачан у овом случају јер смо већ дефинисали да j иде од 1 до i
        # Овај услов би могао и да се изостави
        if i >= j:
            # Исписивање парног броја (2*j) и размака
            # Аргумент end="" у print функцији спречава прелазак у нови ред
            print(2 * j, " ", end="")

        # Инкрементирање бројача колона
        j = j + 1

    # Прелазак у нови ред након исписивања свих елемената у тренутном реду
    # print() без аргумената исписује празан ред
    print()

    # Инкрементирање бројача редова
    i = i + 1

# Резултат извршавања је троугао од n редова где сваки ред i садржи i парних бројева (2, 4, 6, ...)
# Пример за n=4:
# 2
# 2  4
# 2  4  6
# 2  4  6   8

print("\nЈедноставнија имплементација:")
# Коришћење for петље уместо while петље - много једноставнији приступ
# Поново тражимо унос корисника за број редова
n = int(input("Број редова = "))

# Користимо for петљу која је много читљивија од while петље
# range(1, n+1) генерише секвенцу бројева од 1 до n (укључујући n)
# Променљива i представља редни број реда и узима вредности од 1 до n
for i in range(1, n + 1):
    # У једној линији кода радимо следеће:
    # 1. [2*j for j in range(1, i+1)] - листа парних бројева од 2 до 2*i
    #    - range(1, i+1) генерише бројеве од 1 до i
    #    - Сваки број j се множи са 2 да би добили парне бројеве
    # 2. Оператор * (распакивање) пред листом распакује листу у појединачне аргументе
    #    који се прослеђују print функцији
    # 3. print() функција аутоматски додаје размаке између аргумената
    #    и нови ред на крају испис
    print(*[2 * j for j in range(1, i + 1)])

# Резултат је исти као и у претходном примеру, али са значајно мање кода
# и користећи модерне Python функције

print("\nЈедноставан пример коришћења оператора * (распакивање):")
# Креирамо листу бројева
brojevi = [2, 4, 6, 8, 10]

# Исписујемо листу без * оператора - добијамо листу са заградама
print("Исписивање листе без * оператора:")
print(brojevi)  # Излаз: [2, 4, 6, 8, 10]

# Исписујемо листу са * оператором - распакивање елемената
print("Исписивање листе са * оператором:")
print(*brojevi)  # Излаз: 2 4 6 8 10

# Можемо додати и додатне аргументе у print функцију
print("Исписивање са додатним аргументима:")
print(
    "Парни бројеви:", *brojevi, "су лепи бројеви"
)  # Излаз: Парни бројеви: 2 4 6 8 10 су лепи бројеви

# Можемо контролисати сепаратор између распакованих елемената
print("Исписивање са прилагођеним сепаратором:")
print(*brojevi, sep=" | ")  # Излаз: 2 | 4 | 6 | 8 | 10

```
